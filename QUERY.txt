----------TRIGGER----------

**UPPER**

CREATE TRIGGER `case` BEFORE INSERT ON `categories`
 FOR EACH ROW BEGIN
SET NEW.categories_desc=UPPER(NEW.categories_desc);
END

**TIME**

CREATE TRIGGER `time` BEFORE INSERT ON `chef`
 FOR EACH ROW set new.time=now()

**TOTAL**

CREATE TRIGGER `TOTAL` BEFORE INSERT ON `orders`
 FOR EACH ROW SET NEW.TOTAL=NEW.ITEM_RATE*NEW.QUANTITY

CREATE TRIGGER `total_amount1` BEFORE INSERT ON `orders`
 FOR EACH ROW set new.total=new.item_rate*new.quantity



----------STORED PROCEDURE----------

**JOIN**

DELIMITER $$
CREATE PROCEDURE MANAGERS ()
BEGIN
SELECT T. TABLE_ID, T.MGR_ID, C. CUST_NAME, T. TABLE_CAPACITY
FROM TABLE_BOOKING T, CUSTOMERS C
WHERE T.MGR_ID=C.MGR_ID;
END
$$


DELIMITER $$
CREATE PROCEDURE ORDER_BILL ()
BEGIN
SELECT O.ORD_ID,C.CUST_NAME,O.ITEM_ID,M.ITEM_NAME,O.QUANTITY,O.ITEM_RATE,O.TOTAL
FROM ORDERS O, CUSTOMERS C,MENU M
WHERE O.ORD_ID=C.ORD_ID AND M.ITEM_ID=O.ITEM_ID;
END
$$
