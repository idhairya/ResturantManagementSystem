------------------CREATE TABLE-------------------

CREATE TABLE CATEGORIES
(
CATEGORIES_ID VARCHAR(10) PRIMARY KEY,
CATEGORIES_DESC VARCHAR(20)
);



CREATE TABLE MENU
(
ITEM_ID VARCHAR(5) PRIMARY KEY,
ITEM_NAME VARCHAR(20),
ITEM_RATE DOUBLE NOT NULL,
CATEGORIES_ID VARCHAR(5),
FOREIGN KEY(CATEGORIES_ID) REFERENCES CATEGORIES(CATEGORIES_ID) ON DELETE CASCADE
);



CREATE TABLE MANAGER
(
MGR_ID VARCHAR(10) PRIMARY KEY,
NAME VARCHAR(20),
ADDRESS VARCHAR(30),
SALARY FLOAT NOT NULL
);



CREATE TABLE ORDERS
(
ORD_ID VARCHAR(5) PRIMARY KEY,
QUANTITY INT,
ITEM_RATE DOUBLE NOT NULL,
TOTAL DOUBLE NOT NULL,
ITEM_ID VARCHAR(5),
FOREIGN KEY(ITEM_ID) REFERENCES MENU(ITEM_ID) ON DELETE CASCADE
);



CREATE TABLE TABLE_BOOKING
(TABLE_ID VARCHAR(5) PRIMARY KEY,
TABLE_CAPACITY INT,
MGR_ID VARCHAR(10),
FOREIGN KEY(MGR_ID) REFERENCES MANAGER(MGR_ID) ON DELETE CASCADE
);



CREATE TABLE CUSTOMERS
(
CUST_NAME VARCHAR(20),
TABLE_ID VARCHAR(5),
ITEM_ID VARCHAR(5),
ORD_ID VARCHAR(5),
MGR_ID VARCHAR(10),
FOREIGN KEY(TABLE_ID) REFERENCES TABLE_BOOKING(TABLE_ID) ON DELETE CASCADE,
FOREIGN KEY(ITEM_ID) REFERENCES MENU(ITEM_ID) ON DELETE CASCADE,
FOREIGN KEY(ORD_ID) REFERENCES ORDERS(ORD_ID) ON DELETE CASCADE,
FOREIGN KEY(MGR_ID) REFERENCES MANAGER(MGR_ID) ON DELETE CASCADE
);



CREATE TABLE TAKEN_BY
(
ORD_ID VARCHAR(5),
MGR_ID VARCHAR(10),
FOREIGN KEY(ORD_ID) REFERENCES ORDERS(ORD_ID) ON DELETE CASCADE,
FOREIGN KEY(MGR_ID) REFERENCES MANAGER(MGR_ID) ON DELETE CASCADE
);



CREATE TABLE CONTACTS
(
MGR_ID VARCHAR(10),
PHONE_NUM INT(10) PRIMARY KEY,
FOREIGN KEY(MGR_ID) REFERENCES MANAGER(MGR_ID) ON DELETE CASCADE
);



CREATE TABLE CHEF
(
CHEF_NAME VARCHAR(30),
SPECIALITY VARCHAR(30),
CATEGORIES_ID VARCHAR(10),
FOREIGN KEY(CATEGORIES_ID) REFERENCES CATEGORIES(CATEGORIES_ID) ON DELETE CASCADE
);
